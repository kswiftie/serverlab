CREATE TABLE song ( id BIGSERIAL PRIMARY KEY, name VARCHAR(100) ); CREATE TABLE artist ( id BIGSERIAL PRIMARY KEY, name VARCHAR(100) ); CREATE TABLE album ( id BIGSERIAL PRIMARY KEY, name VARCHAR(100) ); CREATE TABLE genre ( id BIGSERIAL PRIMARY KEY, name VARCHAR(100) ); CREATE TABLE client ( id BIGSERIAL PRIMARY KEY, name VARCHAR(100), password TEXT ); CREATE TABLE playlist ( id BIGSERIAL PRIMARY KEY, name VARCHAR(100) ); CREATE TABLE playlist_song ( song_id INT REFERENCES song(id), playlist_id INT REFERENCES playlist(id) ); CREATE TABLE song_listenings ( song_id INT REFERENCES song(id), client_id INT REFERENCES client(id), listenings_count INT ); CREATE TABLE album_song ( song_id INT REFERENCES song(id), album_id INT REFERENCES album(id) ); CREATE TABLE genre_song ( song_id INT REFERENCES song(id), genre_id INT REFERENCES genre(id) ); CREATE TABLE artist_song ( song_id INT REFERENCES song(id), artist_id INT REFERENCES artist(id) ); CREATE TABLE playlist_creator ( playlist_id INT REFERENCES playlist(id), client_id INT REFERENCES client(id) );
